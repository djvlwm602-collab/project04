기존 `dashboard/page.tsx`의 `calculateMonthsToFIRE` 함수를 분리하고, 피드백 문구 선택 로직 추가.

```typescript
// src/lib/calculator.ts
// 역할: FIRE 은퇴 계산 로직 및 피드백 문구 선택
import type { UserProfile } from "./types";
import {
  FIRE_MULTIPLIER,
  SAVING_FEEDBACKS,
  SPENDING_FEEDBACKS,
} from "./constants";

// 목표 자금 계산: 은퇴 후 월 생활비 × 12 × 25(4% 룰)
export function calcTargetAmount(targetExpense: number): number {
  return targetExpense * 12 * FIRE_MULTIPLIER;
}

// 월 저축액 계산
export function calcMonthlySaving(profile: UserProfile): number {
  return profile.monthlyIncome - profile.monthlyExpense;
}

// 복리 기반 FIRE 도달 개월 수 계산
// 최대 600개월(50년) 제한으로 무한 루프 방지
export function calculateMonthsToFIRE(
  assets: number,
  monthlySaving: number,
  target: number,
  annualRate: number
): number {
  const monthlyRate = annualRate / 12;
  let current = assets;
  let months = 0;

  if (current >= target) return 0;
  if (monthlySaving <= 0 && monthlyRate <= 0) return 600;

  while (current < target && months < 600) {
    current = current * (1 + monthlyRate) + monthlySaving;
    months++;
  }
  return months;
}

// UserProfile 기반 은퇴 개월 수 한번에 계산
export function calcMonthsFromProfile(
  profile: UserProfile,
  extraSaving: number = 0
): number {
  const target = calcTargetAmount(profile.targetExpense);
  const saving = calcMonthlySaving(profile) + extraSaving;
  return calculateMonthsToFIRE(
    profile.currentAssets,
    saving,
    target,
    profile.investReturnRate
  );
}

// 슬라이더 피드백 문구 선택 (금액 구간 매칭 + 랜덤)
export function getSliderFeedback(
  sliderValue: number,
  monthDiff: number
): string {
  if (sliderValue === 0) {
    return "현재 페이스로 훌륭하게 저축하고 계십니다! 부장님 면전에 사표 던질 날이 머지않았습니다.";
  }

  const absValue = Math.abs(sliderValue);
  const absDiff = Math.abs(monthDiff);
  const feedbacks = sliderValue > 0 ? SAVING_FEEDBACKS : SPENDING_FEEDBACKS;

  // 금액에 맞는 피드백 항목 찾기 (내림차순이므로 첫 번째 매칭)
  const matched = feedbacks.find((f) => absValue >= f.minAmount);
  if (!matched) return "";

  const msg = matched.messages[Math.floor(Math.random() * matched.messages.length)];
  return msg.replace("{months}", String(absDiff));
}

// 개월 수 → 예상 날짜 문자열 변환
export function formatProjectedDate(months: number): string {
  const d = new Date();
  d.setMonth(d.getMonth() + months);
  return `${d.getFullYear()}년 ${d.getMonth() + 1}월`;
}
```

```typescript
// src/lib/storage.ts
// 역할: localStorage 읽기/쓰기 래퍼 (SSR 안전)
import type { UserProfile, ResistRecord } from "./types";
import { DEFAULT_PROFILE } from "./constants";

const PROFILE_KEY = "bye-company-profile";
const RESIST_KEY = "bye-company-resist-records";
const SETUP_DONE_KEY = "bye-company-setup-done";

// --- UserProfile ---

export function loadProfile(): UserProfile {
  if (typeof window === "undefined") return DEFAULT_PROFILE;
  try {
    const raw = localStorage.getItem(PROFILE_KEY);
    if (!raw) return DEFAULT_PROFILE;
    return { ...DEFAULT_PROFILE, ...JSON.parse(raw) };
  } catch {
    return DEFAULT_PROFILE;
  }
}

export function saveProfile(profile: UserProfile): void {
  if (typeof window === "undefined") return;
  localStorage.setItem(PROFILE_KEY, JSON.stringify(profile));
}

// --- 온보딩 완료 여부 ---

export function isSetupDone(): boolean {
  if (typeof window === "undefined") return false;
  return localStorage.getItem(SETUP_DONE_KEY) === "true";
}

export function markSetupDone(): void {
  if (typeof window === "undefined") return;
  localStorage.setItem(SETUP_DONE_KEY, "true");
}

// --- ResistRecord ---

export function loadResistRecords(): ResistRecord[] {
  if (typeof window === "undefined") return [];
  try {
    const raw = localStorage.getItem(RESIST_KEY);
    if (!raw) return [];
    return JSON.parse(raw);
  } catch {
    return [];
  }
}

export function saveResistRecord(record: ResistRecord): void {
  if (typeof window === "undefined") return;
  const records = loadResistRecords();
  records.unshift(record);
  localStorage.setItem(RESIST_KEY, JSON.stringify(records));
}

// 참기 기록 통계: 총 절약 금액, 총 앞당긴 일수
export function getResistStats(): { totalAmount: number; totalDays: number; count: number } {
  const records = loadResistRecords();
  return {
    totalAmount: records.reduce((sum, r) => sum + r.amount, 0),
    totalDays: records.reduce((sum, r) => sum + r.savedDays, 0),
    count: records.length,
  };
}
```

- [ ] **Step 1: 스트레스 테스트 페이지 생성**

6가지 시나리오 토글 → 적용 시 은퇴일 변동을 실시간 비교 표시.

```tsx
// src/app/stress-test/page.tsx
"use client";

// 역할: 현실적 위기/호재 시나리오를 대입하여 은퇴일 변동을 확인하는 페이지
// 핵심 기능: 6가지 시나리오 토글, 원본 vs 적용 후 은퇴일 비교
// 의존: lib/calculator, lib/storage, lib/constants

import { useState, useMemo, useEffect } from "react";
import { motion } from "framer-motion";
import { useRouter } from "next/navigation";
import { ArrowLeft, ShieldAlert } from "lucide-react";
import type { UserProfile } from "@/lib/types";
import { STRESS_SCENARIOS } from "@/lib/constants";
import { calcMonthsFromProfile, formatProjectedDate } from "@/lib/calculator";
import { loadProfile, isSetupDone } from "@/lib/storage";

export default function StressTestPage() {
  const router = useRouter();
  const [profile, setProfile] = useState<UserProfile | null>(null);
  const [activeIds, setActiveIds] = useState<Set<string>>(new Set());

  useEffect(() => {
    if (!isSetupDone()) {
      router.replace("/dashboard/setup");
      return;
    }
    setProfile(loadProfile());
  }, [router]);

  // 시나리오 토글
  const toggle = (id: string) => {
    setActiveIds((prev) => {
      const next = new Set(prev);
      if (next.has(id)) next.delete(id);
      else next.add(id);
      return next;
    });
  };

  // 원본 은퇴 개월 수
  const baseMonths = useMemo(() => {
    if (!profile) return 0;
    return calcMonthsFromProfile(profile);
  }, [profile]);

  // 시나리오 적용 후 프로필 계산
  const stressedProfile = useMemo(() => {
    if (!profile) return null;
    let result = { ...profile };
    for (const scenario of STRESS_SCENARIOS) {
      if (activeIds.has(scenario.id)) {
        result = scenario.apply(result);
      }
    }
    return result;
  }, [profile, activeIds]);

  const stressedMonths = useMemo(() => {
    if (!stressedProfile) return 0;
    return calcMonthsFromProfile(stressedProfile);
  }, [stressedProfile]);

  const diff = stressedMonths - baseMonths;

  if (!profile) {
    return (
      <div className="flex min-h-screen items-center justify-center bg-background">
        <div className="h-8 w-8 animate-spin rounded-full border-4 border-kakao-yellow border-t-transparent" />
      </div>
    );
  }

  return (
    <div className="flex min-h-screen flex-col bg-background text-foreground">
      {/* 헤더 */}
      <header className="flex h-16 items-center gap-3 border-b border-gray-200 bg-white px-6 shadow-sm dark:border-zinc-800 dark:bg-card">
        <button
          onClick={() => router.push("/dashboard")}
          className="rounded-xl p-2 text-subtext hover:bg-gray-100 dark:hover:bg-zinc-800"
        >
          <ArrowLeft size={20} />
        </button>
        <h1 className="text-lg font-extrabold tracking-tight">스트레스 테스트</h1>
      </header>

      <main className="flex flex-1 flex-col items-center p-6 sm:p-10">
        <div className="w-full max-w-md space-y-6">

          {/* 비교 결과 카드 */}
          <motion.div
            initial={{ opacity: 0, y: 20 }}
            animate={{ opacity: 1, y: 0 }}
            className="rounded-[32px] bg-card p-8 shadow-[0_8px_30px_rgb(0,0,0,0.06)] dark:shadow-[0_8px_30px_rgb(0,0,0,0.2)]"
          >
            <div className="flex items-center gap-2 mb-6">
              <ShieldAlert size={22} className="text-kakao-brown dark:text-kakao-yellow" />
              <h2 className="text-[17px] font-extrabold">시나리오 적용 결과</h2>
            </div>

            <div className="grid grid-cols-2 gap-4 text-center">
              <div className="rounded-2xl bg-gray-50 p-4 dark:bg-zinc-800">
                <p className="text-[12px] text-subtext font-medium mb-1">현재 예상</p>
                <p className="text-xl font-extrabold">{formatProjectedDate(baseMonths)}</p>
                <p className="text-[12px] text-subtext mt-1">{baseMonths}개월</p>
              </div>
              <div className={`rounded-2xl p-4 ${
                activeIds.size === 0
                  ? "bg-gray-50 dark:bg-zinc-800"
                  : diff > 0
                  ? "bg-red-50 dark:bg-red-900/20"
                  : "bg-blue-50 dark:bg-blue-900/20"
              }`}>
                <p className="text-[12px] text-subtext font-medium mb-1">시나리오 적용</p>
                <p className="text-xl font-extrabold">
                  {activeIds.size === 0 ? "—" : formatProjectedDate(stressedMonths)}
                </p>
                {activeIds.size > 0 && (
                  <p className={`text-[12px] mt-1 font-bold ${
                    diff > 0 ? "text-red-500" : "text-blue-500"
                  }`}>
                    {diff > 0 ? `+${diff}개월 늘어남` : diff < 0 ? `${diff}개월 단축!` : "변동 없음"}
                  </p>
                )}
              </div>
            </div>

            {stressedMonths >= 600 && activeIds.size > 0 && (
              <div className="mt-4 rounded-xl bg-red-100 p-3 text-center text-[13px] font-bold text-red-700 dark:bg-red-900/30 dark:text-red-300">
                이 시나리오에서는 50년 내 은퇴가 어렵습니다
              </div>
            )}
          </motion.div>

          {/* 시나리오 목록 */}
          <div className="space-y-3">
            <p className="text-[13px] font-bold text-subtext px-1">시나리오를 선택하세요 (복수 선택 가능)</p>
            {STRESS_SCENARIOS.map((scenario, i) => {
              const isActive = activeIds.has(scenario.id);
              return (
                <motion.button
                  key={scenario.id}
                  initial={{ opacity: 0, y: 10 }}
                  animate={{ opacity: 1, y: 0 }}
                  transition={{ delay: i * 0.06 }}
                  onClick={() => toggle(scenario.id)}
                  className={`w-full rounded-2xl p-5 text-left transition-all ${
                    isActive
                      ? "bg-kakao-yellow/20 border-2 border-kakao-yellow dark:bg-kakao-yellow/10"
                      : "bg-card border-2 border-transparent shadow-sm dark:shadow-[0_4px_20px_rgb(0,0,0,0.15)]"
                  }`}
                >
                  <div className="flex items-center gap-3">
                    <span className="text-2xl">{scenario.emoji}</span>
                    <div className="flex-1">
                      <p className="text-[15px] font-extrabold text-foreground">{scenario.name}</p>
                      <p className="text-[12px] text-subtext mt-0.5">{scenario.description}</p>
                    </div>
                    <div className={`h-5 w-5 rounded-full border-2 transition-colors ${
                      isActive
                        ? "bg-kakao-yellow border-kakao-yellow"
                        : "border-gray-300 dark:border-zinc-600"
                    }`} />
                  </div>
                </motion.button>
              );
            })}
          </div>

          {/* 초기화 버튼 */}
          {activeIds.size > 0 && (
            <button
              onClick={() => setActiveIds(new Set())}
              className="w-full rounded-[16px] border border-gray-200 py-3 text-[14px] font-bold text-subtext dark:border-zinc-700"
            >
              전체 초기화
            </button>
          )}
        </div>
      </main>
    </div>
  );
}
```
